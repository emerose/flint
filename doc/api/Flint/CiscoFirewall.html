223:       if input[0].kind_of? PortRange
224:         r = (_resolve(input[0].start,prot).._resolve(input[0].stop,prot))
225:       elsif input[0].kind_of? PortNumber
226:         r = _resolve(input[0],prot)
227:       elsif input[0].kind_of? PortName
228:         r = _resolv(input[0].name,prot)
229:       elsif input[0].kind_of? String
230:         if input[0].match(/^eq$/i)
231:           r = _resolve(input[1],prot)
232:         elsif input[0].match(/^range$/i)
233:           r = (_resolve(input[1].start,prot).._resolve(input[1].stop,prot))
234:         elsif input[0].match(/^lt$/i)
235:           r = (0.._resolve(input[1],prot))
236:         elsif input[0].match(/^gt$/i)
237:           r = (_resolve(input[1],prot)..65535)
238:         elsif input[0].match(/^neq$/i)
239:           r = [(0.._resolve(input[1],prot) - 1),
240:                ((_resolve(input[1],prot) + 1)..65535)]
241:           def r.exclusion?
242:             true
243:           end
244:         end
245:       elsif input[0] == :object_group_ref

249:           (r = group.members.map do |x|

253:           []
254:         end

258:       end

 35:           end

 37:         end
 38:         save
 39:       end

 41:       # gotta build our name and DB first
 42:       name_pattern = /^name$/i

 46:           if (ast = line.load_ast) and ast.first.match(name_pattern)
 47:             resolver.name(ast[2], ast[1])
 48:           end
 49:         end
 50:       end

 53:         lines.all.each do |line|
 54:           next unless (ast = line.load_ast)

 59:
